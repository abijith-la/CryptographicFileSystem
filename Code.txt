	choice = input("Would you like to (E)encrypt or (D)Decrypt ")

	if choice == 'E':
		filename = input("File to encrypt: ")
		encrypt(key, filename)     
		print('Done.')
	elif choice == 'D':
		filename = input("File to decrypt: ")
		decrypt(key, filename)
		print("Done.")

	else:
		print("No option selected, closing...")
